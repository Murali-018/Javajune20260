Creation of Table patient :
CREATE TABLE Patient (patient_id INT PRIMARY KEY,patient_name VARCHAR(50),age INT,gender VARCHAR(10),phone VARCHAR(15));

Creation of Table Doctor: 
CREATE TABLE Doctor (doctor_id INT PRIMARY KEY, doctor_name VARCHAR(50),specialization VARCHAR(50),phone VARCHAR(15));

Creation of Table Appointment:
CREATE TABLE Appointment (appointment_id INT PRIMARY KEY,patient_id INT,doctor_id INT,appointment_date DATE,FOREIGN KEY (patient_id) REFERENCES Patient(patient_id),
FOREIGN KEY (doctor_id) REFERENCES Doctor(doctor_id));

Subquery:
1.Find patients who have appointments with a specific doctor.
SELECT *
FROM Patient
WHERE patient_id IN (
    SELECT patient_id
    FROM Appointment
    WHERE doctor_id = (
        SELECT doctor_id
        FROM Doctor
        WHERE doctor_name = 'Dr. Siva'
    )
);

2.Display doctors who have at least one appointment.
SELECT *
FROM Doctor d
WHERE EXISTS (
    SELECT *
    FROM Appointment a
    WHERE d.doctor_id = a.doctor_id
);

3.Find patients who have never booked an appointment.
SELECT *
FROM Patient
WHERE patient_id NOT IN (
    SELECT patient_id
    FROM Appointment
);

4.Display patients who have at least one appointment
SELECT *
FROM Patient p
WHERE EXISTS (
    SELECT *
    FROM Appointment a
    WHERE p.patient_id = a.patient_id
);


Correlated subquery
1.Find doctors who have more than one appointment.
SELECT *
FROM Doctor d
WHERE (
    SELECT COUNT(*)
    FROM Appointment a
    WHERE a.doctor_id = d.doctor_id
) > 1;

2.Find doctors who have treated female patients
SELECT *
FROM Doctor d
WHERE EXISTS (
    SELECT *
    FROM Appointment a
    JOIN Patient p
    ON a.patient_id = p.patient_id
    WHERE a.doctor_id = d.doctor_id
      AND p.gender = 'Female'
);
